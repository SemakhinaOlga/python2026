bookings = [('designers', 8), ('marketing', 5), ('designers', 4), ('startups', 6)]
finished = ['marketing']

hours = {}
for team, time in bookings:
    hours[team] = hours.get(team, 0) + time

for team in finished:
    removed = hours.pop(team, '0 часов')
    print(f"Команда {team} удалена, было забронировано: {removed}")

print("Обновлённые записи:", hours)
